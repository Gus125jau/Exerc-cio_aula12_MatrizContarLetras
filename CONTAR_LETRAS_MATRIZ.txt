Algoritmo "Contar_Letras_Matriz"

Var
    matriz: vetor[1..3, 1..3] de caractere
    i, j: inteiro
    letra: caractere
    contador: inteiro

Inicio
    contador <- 0

    // O usuário informa as leituras para ser realizada a leitura da matriz
    EscrevaL("Digite os elementos da matriz 3x3 (apenas letras):", "")
    Para i de 1 ate 3 Faça
        Para j de 1 ate 3 Faça
            Escreva("Matriz[", i, "][", j, "]: ")
            Leia(matriz[i, j])
        FimPara
    FimPara

    // Leitura da letra a ser contada que foi escolhida pelo usuário
    Escreva("Digite a letra que deseja contar: ")
    Leia(letra)

    // Contagem das ocorrências da letra, ou seja, número de repetições
    Para i de 1 ate 3 Faça
        Para j de 1 ate 3 Faça
            Se matriz[i, j] = letra Entao
                contador <- contador + 1
            FimSe
        FimPara
    FimPara

    // Informação do Resultado do resultado
    Escreval("A letra '", letra, "' aparece ", contador, " vezes na matriz.")

Fimalgoritmo
